Detect Cycle in a Graph

        bool dfs(vector<vector<int>>& graph,vector<int> &visited,vector<int>&inpath,int i){
          visited[i]=1;
          inpath[i]=1;
          for(auto it:graph[i]){
              if(!visited[it] && dfs(graph,visited,inpath,it) ){
                  return true;
              }
              else if(inpath[it]){
                  return true;
              }
          }
          inpath[i]=0;
          return false;
      }
